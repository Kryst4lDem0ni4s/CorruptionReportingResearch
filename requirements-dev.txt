# ============================================================================
# USAGE GUIDELINES
# ============================================================================
#
# 1. Install production dependencies first:
#    pip install -r requirements.txt
#
# 2. Then install development dependencies:
#    pip install -r requirements-dev.txt
#
# 3. Or install everything at once:
#    pip install -r requirements-dev.txt
#    (This includes production dependencies via -r requirements.txt)
#
# 4. Set up pre-commit hooks:
#    pre-commit install
#
# 5. Run tests:
#    pytest tests/
#
# 6. Check code quality:
#    black --check backend/
#    flake8 backend/
#    mypy backend/
#
# 7. Generate coverage report:
#    pytest --cov=backend --cov-report=html
#
# 8. Profile code:
#    python -m memory_profiler backend/main.py
#
# ============================================================================

# ============================================================================
# RECOMMENDED DEVELOPMENT WORKFLOW
# ============================================================================
#
# 1. Create virtual environment:
#    python -m venv venv
#    source venv/bin/activate  # Linux/Mac
#    venv\Scripts\activate     # Windows
#
# 2. Install dependencies:
#    pip install --upgrade pip
#    pip install -r requirements-dev.txt
#
# 3. Download ML models:
#    python scripts/download_models.py
#
# 4. Initialize storage:
#    python scripts/initialize_storage.py
#
# 5. Run tests:
#    pytest tests/ -v --cov=backend
#
# 6. Start development server:
#    uvicorn backend.main:app --reload
#
# 7. Access API docs:
#    http://localhost:8000/docs
#
# ============================================================================

# ============================================================================
# PRE-COMMIT CONFIGURATION
# ============================================================================
#
# Create .pre-commit-config.yaml:
#
# repos:
#   - repo: https://github.com/psf/black
#     rev: 24.1.1
#     hooks:
#       - id: black
#
#   - repo: https://github.com/pycqa/flake8
#     rev: 7.0.0
#     hooks:
#       - id: flake8
#
#   - repo: https://github.com/pycqa/isort
#     rev: 5.13.2
#     hooks:
#       - id: isort
#
#   - repo: https://github.com/pre-commit/mirrors-mypy
#     rev: v1.8.0
#     hooks:
#       - id: mypy
#
# Then run: pre-commit install
#
# ============================================================================
